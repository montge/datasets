527         private JScrollPane getJScrollPane() {

529                 if (null == tableBC) {
530                     tableModel = new SitoTableModel();
531                     tableBC = new ScrollableTable(tableModel);
532                     tableBC.setRowHeight(32);

534                             new MediaVotiRenderer());
535                     tableBC.setDefaultRenderer(Punto3D.class,
536                             new Punto3DRenderer());
537                     tableBC.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
538                     tableBC.setSelectionForeground(Color.RED);
539                     tableBC.setSelectionBackground(Color.white);
540                     tableBC.setShowVerticalLines(false);
541                     tableBC.setColumnSelectionAllowed(false);
542                     tableBC.addMouseListener(bcHelp);
543                     tableBC.setName("tableBC");
544                     /**
545                      * SelectionListener - if a selected row, the buttons
546                      * Tab, edit and delete are active. Otherwise, are
547                      * Disabled.
548                      */
549                     ListSelectionModel selectionModel = tableBC.getSelectionModel();
550                     selectionModel.addListSelectionListener(new ListSelectionListener() {
551                         public void valueChanged(ListSelectionEvent pEvent) {
552                             if (tableBC.getSelectedRow() != -1) {
553                                 btnEliminaBC.setEnabled(true);
554                                 btnSchedaBC.setEnabled(true);
555                                 btnModificaBC.setEnabled(true);
556                             } else {
557                                 btnEliminaBC.setEnabled(false);
558                                 btnSchedaBC.setEnabled(false);
559                                 btnModificaBC.setEnabled(false);
560                             }
561                         }
562                     });
563                     /**
564                      * KeyListener <ENTER> - Details of the cultural selected.
565                      * <Backspace> - Delete the selected cultural. <space> --
566                      * Modify the cultural selected.
567                      */
568                     tableBC.addKeyListener(new KeyAdapter() {
569                         public void keyPressed(KeyEvent pEvent) {
570                             int keyCode = pEvent.getKeyCode();
571                             if (keyCode == KeyEvent.VK_ENTER) {
572                                 btnSchedaBC.doClick();
573                             } else if (keyCode == KeyEvent.VK_BACK_SPACE) {
574                                 btnEliminaBC.doClick();
575                             } else if (keyCode == KeyEvent.VK_SPACE) {
576                                 btnModificaBC.doClick();
577                             }
578                         }
579                     });
580                 }
581     
582                 JScrollPane = new JScrollPane(tableBC);
583                 JScrollPane.setHorizontalScrollBarPolicy(JScrollPane.HORIZONTAL_SCROLLBAR_ALWAYS);
584                 JScrollPane.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
585             }
586             return JScrollPane;
587         }

650             SchedaBC nuova;
651             if (pbc == null) {
652                 nuova = new SchedaBC(this);
653             } else {
654                 for (int i = 0; i < figli.size(); i++) {
655                     SchedaBC corrente = figli.get(i);
656                     if (pbc.getId() == corrente.getId()) {

663             jDesktopPane.add(nuova, Integer.MAX_VALUE);
664             desktopManager.centerFrame(nuova);
665             nuova.setVisible(true);
666             figli.add(nuova);
