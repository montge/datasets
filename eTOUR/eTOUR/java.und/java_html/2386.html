52         /**
53          * Delete a banner and the image associated with the system.
54          *
55          * @param pIdBanner id of the banner to remove
56          * @throws RemoteException
57          */
58         public boolean cancellaBanner(int pIdBanner) throws RemoteException {
59             /* Bean containing the data of the banner */
60             BeanBanner banner;
61     
62             try {
63                 /* Load the banner and check the data */
64                 banner = dbBanner.ottieniBannerDaID(pIdBanner);
65                 if (!ControlloDati.checkBeanBanner(banner)) {
66                     throw new RemoteException(MessaggiErrore.ERRORE_DATI);
67                 }
68                 /* Remove the image associated with the banner */
69                 File immagine = new File(banner.getPercorsoFile());
70                 immagine.delete();
71                 /* Clear the whole bean banner */
72                 return (dbBanner.cancellaBanner(pIdBanner));
73             }
74             catch (SQLException e) {
75                 throw new RemoteException(MessaggiErrore.ERRORE_CONNESSIONE_DBMS);
76             }
77         }

79         /**
80          * Method to change the image associated with the banner:
81          * Delete the old image and stores the new
82          * Enjoy the same path as the old image.
83          *
84          * @param pBannerID id banner
85          * @param pImmagine ImageIcon object containing the new image of the banner
86          * @return returns true if the operation is successful
87          */
88         public boolean modificaBanner(int pBannerID, ImageIcon pImmagine) throws RemoteException {
89             try {
90                 /* Performs a check on the image of the banner */
91                 if (!ControlloDati.checkImmagine(pImmagine)) {
92                     throw new RemoteException(MessaggiErrore.ERRORE_DATI);
93                 }
94                 /* Load the icon image */
95                 BufferedImage buffImg = (BufferedImage) pImmagine.getImage();
96                 /* Load the data of the banner */
97                 BeanBanner banner = dbBanner.ottieniBannerDaID(pBannerID);
98                 /* Check the banner on the data uploaded */
99                 if (!ControlloDati.checkBeanBanner(banner)) {
100                     throw new RemoteException(MessaggiErrore.ERRORE_DATI);
101                 }
102                 /* Rewrite the image file and returns the result of the operation */
103                 File Imgfile = new File(banner.getPercorsoFile());
104                 return (ImageIO.write(buffImg, "jpg", Imgfile));
105             }
106             catch (SQLException e) {
107                 throw new RemoteException(MessaggiErrore.ERRORE_CONNESSIONE_DBMS);
108             }
109             catch (Exception e) {
110                 if (e instanceof RemoteException) {
111                     throw new RemoteException(MessaggiErrore.ERRORE_DATI);
112                 } else {
113                     throw new RemoteException(MessaggiErrore.ERRORE_FILE);
114                 }
115             }
116         }

118         /**
119          * Returns a list of banners for a refreshment.
120          * Use the method (@link) DBBanner.ottieniBanner
121          *
122          * @param pIdPuntoDiRistoro ID refreshment
123          * @throws RemoteException
124          */
125         public HashMap<BeanBanner, ImageIcon> ottieniBannersDaID(int pIdPuntoDiRistoro) throws RemoteException {
126             /* Create a HashMap that will hold the banner and associated images */
127             HashMap<BeanBanner, ImageIcon> toReturn = new HashMap<BeanBanner, ImageIcon>();
128     
129             try {
130                 /* Get the list of banners for a refreshment point */
131                 ArrayList<BeanBanner> listaBanner = dbBanner.ottieniBanner(pIdPuntoDiRistoro);
132                 /* For each banner valid list */
133                 for (BeanBanner banner : listaBanner) {
134                     /* If the banner is a valid charge in the HashMap with the image file */
135                     if (ControlloDati.checkBeanBanner(banner)) {
136                         /* Create the image file */
137                         File fileImmagine = new File(banner.getPercorsoFile());
138                         /* Create an object ImageIcon from the image file */
139                         ImageIcon icon = new ImageIcon(ImageIO.read(fileImmagine));
140                         /* Add the banner and the image all'HashMap */
141                         toReturn.put(banner, icon);
142                     }
143                 }
144                 return toReturn;
145             }
146             catch (SQLException e) {
147                 throw new RemoteException(MessaggiErrore.ERRORE_CONNESSIONE_DBMS);
148             }
149             catch (IOException e) {
150                 throw new RemoteException(MessaggiErrore.ERRORE_FILE);
151             }
152             catch (Exception e) {
153                 throw new RemoteException(MessaggiErrore.ERRORE_SCONOSCIUTO);
154             }
155         }

157         /**
158          * Create and insert a new banner, making sure the num. maximum
159          * Banner displayed for a refreshment point is
160          * Already been reached in this case throws an exception to the calling method.
161          *
162          * @param pIdPuntoDiRistoro Id of refreshment
163          * @param pImmagineBanner ImageIcon object containing the banner image
164          * @return Returns true if the insertion has been successful.
165          * @exception RemoteException
166          */
167         public boolean inserisciBanner(int pIdPuntoDiRistoro, ImageIcon pImmagineBanner) throws RemoteException {
168             try {
169                 /* Performs a check on the image */
170                 if (!ControlloDati.checkImmagine(pImmagineBanner)) {
171                     throw new RemoteException(MessaggiErrore.ERRORE_DATI);
172                 }
173                 /* Create an object that handles the conventions */
174                 IDBConvenzione dbConvenzione = new DBConvenzione();
175                 /* Load the data of the convention of a refreshment point */
176                 BeanConvenzione convenzione = dbConvenzione.ottieniConvezioneAttiva(pIdPuntoDiRistoro);
177                 /* Get the number of banners displayed by the Convention Masssimo */
178                 int maxBanner = convenzione.getMaxBanner();
179                 /* Load the list of banners for a refreshment and stores the number */
180                 int numBanner = (dbBanner.ottieniBanner(pIdPuntoDiRistoro)).size();
181                 /* Check that the maximum number of banners is not reached */
182                 if (!(numBanner < maxBanner)) {
183                     throw new Exception(MessaggiErrore.ERRORE_NUM_BANNER);
184                 }
185                 /* Create a blank banner */
186                 BeanBanner banner = new BeanBanner();
187                 /* Create a unique file name */
188                 String path = CostantiGlobali.SERVER_IMAGE_PATH + pIdPuntoDiRistoro;
189                 int i = 0;
190                 File fileImg = new File(path + "_" + i + ".jpg");
191                 while (fileImg.exists()) {
192                     i++;
193                     fileImg = new File(path + "_" + i + ".jpg");
194                 }
195                 /* Stores the image in the file system */
196                 BufferedImage im = (BufferedImage) pImmagineBanner.getImage();
197                 if (!ImageIO.write(im, "jpg", fileImg)) {
198                     /* Write failed */
199                     throw new IOException();
200                 }
201                 /* Loads the data in the banner */
202                 banner.setIdPuntoDiRistoro(pIdPuntoDiRistoro);
203                 banner.setPercorsoFile(fileImg.getPath());
204                 /* Insert the banner in the database */
205                 return (dbBanner.inserisciBanner(banner));
206             }
207             catch (SQLException e) {
208                 throw new RemoteException(MessaggiErrore.ERRORE_CONNESSIONE_DBMS);
209             }
210             catch (IOException e) {
211                 throw new RemoteException(MessaggiErrore.ERRORE_FILE);
212             }
213             catch (Exception e) {
214                 if (e.getMessage().equals(MessaggiErrore.ERRORE_NUM_BANNER)) {
215                     /* Has been reached on most of banner inserted num */
216                     throw new RemoteException(MessaggiErrore.ERRORE_NUM_BANNER);
217                 } else {
218                     throw new RemoteException(MessaggiErrore.ERRORE_SCONOSCIUTO);
219                 }
220             }
221         }
