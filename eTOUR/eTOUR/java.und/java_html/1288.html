21          * Please formal control and consistency on
22          * Data content of the banner in the bean passed by parameter.
23          * @param pBanner bean contains the data of the banner.
24          * @return
25          */
26         public static boolean controlloBanner(BeanBanner pBanner) {
27             boolean toReturn = false;
28     
29             if (pBanner != null) {
30                 toReturn = (pBanner.getId() > 0 && pBanner.getPercorsoFile() != ""
31                             && pBanner.getIdPuntoDiRistoro() > 0);
32             }
33     
34             return toReturn;
35         }

37         /**
38          * Method that performs consistency checks and
39          * Correctness of the information contained in the bean past
40          * Per parameter, in particular check that the dates
41          * Publication and expiration of the news are consistent,
42          * Or that the second is the later.
43          *
44          * @param pNews bean containing data news
45          * @return Returns true if the bean contains consistent data
46          */
47         public static boolean controlloNews(BeanNews pNews) {
48             boolean toReturn = false;
49     
50             /* Check the validity of the general method parameter */
51             if (pNews != null) {
52     
53                 Date dataPubb = pNews.getDataPubblicazione(); // Released
54                 Date dataScad = pNews.getDataScadenza(); // Due Date
55                 String news = pNews.getNews(); // Text of News
56     
57                 /* Checking the invalidity of the fields */
58                 if (dataPubb != null && dataScad != null && news != null) {
59                     /* Check the consistency of the dates */
60                     toReturn = dataPubb.before(dataScad);
61     
62                     /* Check that the text is not empty */
63                     toReturn = toReturn && (news != "");
64     
65                     /* Check that the ID is greater than 0 */
66                     toReturn = toReturn && (pNews.getId() > 0);
67     
68                     /* Check the priority value */
69                 }
70             }
71     
72             return toReturn;
73         }
