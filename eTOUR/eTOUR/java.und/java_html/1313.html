15     public class ControlloVisiteBeniCulturali {
16         /**
17          * Please consistency check by calling the appropriate methods
18          *
19          * @param pVBC The cultural visit to check
20          * @return boolean The result of the check: true if OK, false otherwise
21          */
22         public static boolean controllaDatiVisitaBeneCulturale(BeanVisitaBC pVBC) {
23             // If the bean is null
24             if (pVBC == null || !(pVBC instanceof BeanVisitaBC))
25                 return false;
26     
27             // Check the ID of a cultural visit
28             // And the id of its tourist
29             if (!(pVBC.getIdBeneCulturale() > 0) || !(pVBC.getIdTurista() > 0))
30                 return false;
31     
32             // Check the vote (of course ratings are accepted only between 1 and 5
33             if (!(pVBC.getVoto() >= 1 && pVBC.getVoto() <= 5))
34                 return false;
35     
36             // Check for null references in the bean
37             if (!controllaDatiNulli(pVBC))
38                 return false;
39     
40             // Check the correct length of string
41             if (!(pVBC.getCommento().length() > 0))
42                 return false;
43     
44             return true;
45         }
46     
47         /**
48          * Check for null data in a bean cultural visit
49          *
50          * @param pBC The cultural visit to check
51          * @return boolean The result of the check: true if there are no null references, false otherwise
52          */
53         public static boolean controllaDatiNulli(BeanVisitaBC pBC) {
54             // Check the nullity of the fields of feedback
55             if (pBC.getCommento() == null || pBC.getDataVisita() == null)
56                 return false;
57     
58             return true;
59         }
