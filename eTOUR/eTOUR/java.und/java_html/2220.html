33     public class FeedBackTableModel extends AbstractTableModel {
34     
35         private static final long serialVersionUID = 1L;
36         private static final String[] headers = {"rating", "Comment", "Release Date", "Issued by"};
37         private static final Class[] columnClasses = {Integer.class, String.class, Date.class, String.class};
38         private ArrayList<Object[]> data;
39     
40         /**
41          * Default Constructor. We only provide the model but not
42          * Loads no data in it.
43          */
44         public FeedBackTableModel() {
45             data = new ArrayList<Object[]>();
46         }
47     
48         /**
49          * Constructor that takes as input an ArrayList of BeanVisitaBC or
50          * BeanVisitaPR and copies of the data within the model
51          * Preparing for display.
52          *
53          * @param pDatiFeedback java.util.ArrayList - the ArrayList of bean.
54          */
55         public FeedBackTableModel(HashMap<?, String> pDatiFeedback) {
56             this();
57             if (pDatiFeedback == null || pDatiFeedback.size() == 0) {
58                 return;
59             }
60             Iterator<?> itera = pDatiFeedback.keySet().iterator();
61             while (itera.hasNext()) {
62                 Object current = itera.next();
63                 if (current instanceof BeanVisitaBC) {
64                     insertVisitaBC((BeanVisitaBC) current, pDatiFeedback.get(current));
65                 }
66                 else if (current instanceof BeanVisitaPR) {
67                     insertVisitaPR((BeanVisitaPR) current, pDatiFeedback.get(current));
68                 }
69             }
70         }

90         /**

92          *
93          * @return String - the name of the column.
94          * @exception IllegalArgumentException - if the column index is not provided
95          * In the model.
96          */
97         public String getColumnName(int pColumn) throws IllegalArgumentException {
98             if (pColumn >= getColumnCount() || pColumn < 0) {
99                 throw new IllegalArgumentException(
100                     "The column index is not provided in the model.");
101             }
102             return headers[pColumn];

105         /**
106          * Returns the object in the model to the row and column provided.
107          *
108          * @param pRow - the line number.
109          * @param pColumn - the column number.
110          * @return Object - the object contained in the selected cell.
111          * @exception IllegalArgumentException - if the index or the row or column not provided
112          * Are present in the model.
113          */
114         public Object getValueAt(int pRow, int pColumn) throws IllegalArgumentException {
115             if (pRow >= getRowCount() || pRow < 0) {
116                 throw new IllegalArgumentException("The row index is not provided in the model.");
117             }
118             if (pColumn >= getColumnCount() || pColumn < 0) {
119                 throw new IllegalArgumentException("The column index is not provided in the model.");

121             return data.get(pRow)[pColumn];

161         /**
162          * Enter data on the feedback received from a cultural
163          * In the model from Bean.
164          *
165          * @param pVisitaBC BeanVisitaBC - the bean that contains the feedback from
166          * Cultural.
167          * @throws IllegalArgumentException - if the parameters supplied input not
168          * Are valid.
169          */
170         public void insertVisitaBC(BeanVisitaBC pVisitaBC, String pUsername) throws IllegalArgumentException {
171             if (pVisitaBC == null || pUsername == null || pUsername.equals("")) {
172                 throw new IllegalArgumentException("invalid parameters supplied input.");
173             }
174             Object[] aRow = new Object[6];
175             aRow[0] = pVisitaBC.getVoto();
176             aRow[1] = pVisitaBC.getCommento();
177             aRow[2] = pVisitaBC.getDataVisita();
178             aRow[3] = pUsername;
179             aRow[4] = pVisitaBC.getIdBeneCulturale();
180             aRow[5] = pVisitaBC.getIdTurista();
181             data.add(aRow);
182         }

184         /**
185          * Enter data on the feedback received from a refreshment
186          * In the model from Bean.
187          *

189          * Refreshment.
190          * @throws IllegalArgumentException - if the parameters supplied input not
191          * Are valid.

193         public void insertVisitaPR(BeanVisitaPR pVisitaPR, String pUsername) throws IllegalArgumentException {
194             if (pVisitaPR == null || pUsername == null || pUsername.equals("")) {
195                 throw new IllegalArgumentException("invalid parameters supplied input.");
196             }
197             Object[] aRow = new Object[6];
198             aRow[0] = pVisitaPR.getVoto();
199             aRow[1] = pVisitaPR.getCommento();
200             aRow[2] = pVisitaPR.getDataVisita();
201             aRow[3] = pUsername;
202             aRow[4] = pVisitaPR.getIdPuntoDiRistoro();
203             aRow[5] = pVisitaPR.getIdTurista();
204             data.add(aRow);

207         /**
208          * Updates the comment feedback contained in the table row selected.
209          *
210          * @param pNuovoCommento String - the new comment.
211          * @param pRow - the row to update.
212          * @throws IllegalArgumentException - if <ul> <li> the row index is not present in the model.
213          * <li> New comment supplied input is zero. </ul>
214          */
215         public void modificaCommento(String pNuovoCommento, int pRow) throws IllegalArgumentException {
216             if (pRow >= getRowCount() || pRow < 0) {
217                 throw new IllegalArgumentException("The row index is not provided in the model.");
218             }
219             if (null == pNuovoCommento) {
220                 throw new IllegalArgumentException("The new comment can not be null.");
221             }
222             data.get(pRow)[1] = pNuovoCommento;
223             fireTableDataChanged();
224         }

226         /**
227          * Returns the id of the row receive feedback provided on input.
228          *
229          * @param pRow int - the row number.
230          * @return int[] - the id of the feedback.
231          * @throws IllegalArgumentException - if the row index does not pro
232          * In the model.
233          */
234         public int[] getIDFeedback(int pRow) throws IllegalArgumentException {
235             if (pRow >= getRowCount() || pRow < 0) {
236                 throw new IllegalArgumentException("The row index is not provided in the model.");
237             }
238             int[] ids = new int[2];
239             ids[0] = (Integer) data.get(pRow)[4];
240             ids[1] = (Integer) data.get(pRow)[5];
241             return ids;
242         }
243     
244         /**
245          * Returns the id of feedback to line number provided as input and removes it from the model.
246          *
247          * @param pRow int - the row number.
248          * @return int[] - the id of the feedback.
249          * @exception IllegalArgumentException - if the row index does not pro
250          * In the model.
251          */
252         public int[] removeFeedback(int pRow) throws IllegalArgumentException {
253             int[] ids = getIDFeedback(pRow);
254             data.remove(pRow);
255             return ids;
256         }
