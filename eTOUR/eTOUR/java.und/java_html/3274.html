690                                                 .length() + 1)));

692                         }
693                     }
694     
695                     public void keyPressed(KeyEvent pKeyEvent) {
696                         int keyCode = pKeyEvent.getKeyCode();
697                         if (keyCode == KeyEvent.VK_CANCEL
698                                 || keyCode == KeyEvent.VK_BACK_SPACE) {

701                                 labelCaratteri.setText("# Characters"
702                                         + (200 - len + 1));

705                     }
706                 });
