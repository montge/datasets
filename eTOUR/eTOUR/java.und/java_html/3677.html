86         /**
87          * Calculate the distance between the point and another point given as argument
88          *
89          * @Param p
90          * @Return
91          */
92         public double distance(Punto3D p) {
93             double differenzaLongitudine = this.longitude - p.longitude;
94             double p1 = Math.pow(Math.cos(p.latitude)
95                     * Math.sin(differenzaLongitudine), 2);
96             double p2 = Math.pow(Math.cos(latitude) * Math.sin(p.latitude)
97                     - Math.sin(latitude) * Math.cos(p.latitude)
98                     * Math.cos(differenzaLongitudine), 2);
99             double p3 = Math.sin(latitude) * Math.sin(p.latitude)
100                     + Math.cos(latitude) * Math.cos(p.latitude)
101                     * Math.cos(differenzaLongitudine);
102             return (Math.atan(Math.sqrt(p1 + p2) / p3) * EARTH_RADIUS);
103         }

114         public static Punto3D gradiRadianti(double pLatitudine,

116             Punto3D punto = new Punto3D();
