26     public class BannerDialog extends JPanel {

32         /**
33          * This is the default constructor.

35         public BannerDialog() {
36             super(null);
37             this.that = this;
38             setPreferredSize(new Dimension(420, 160));
39             anteprima = new JLabel();
40             anteprima.setBounds(new Rectangle(40, 20, 250, 60));
41             anteprima.setBorder(BorderFactory.createTitledBorder(BorderFactory
42                     .createLineBorder(new Color(51, 102, 255), 3),
43                     "Preview Banner", TitledBorder.DEFAULT_JUSTIFICATION,
44                     TitledBorder.DEFAULT_POSITION,
45                     new Font("Dialog", Font.BOLD, 12), new Color(0, 102, 204)));
46             btnCarica = new JButton();
47             btnCarica.setBounds(new Rectangle(320, 30, 50, 40));
48             btnCarica.setIcon(new ImageIcon(getClass().getResource(
49                     Home.URL_IMAGES + "ApriFile.png")));
50             btnCarica.addActionListener(new ActionListener() {
51                 public void actionPerformed(ActionEvent arg0) {
52                     File f = apriFileDialog();
53     
54                     if (f == null) {
55                         return;
56                     }
57     
58                     BufferedImage img = null;
59                     try {
60                         img = ImageIO.read(f);
61                         if (img.getWidth() > bannerSize.width
62                                 || img.getHeight() > bannerSize.height) {
63                             img = img.getSubimage(0, 0, bannerSize.width,
64                                     bannerSize.height);
65                         }
66                     } catch (Exception e) {
67                         e.printStackTrace();
68                     }
69                     ImageIcon n = new ImageIcon(img);
70                     anteprima.setIcon(n);
71                     anteprima.repaint();
72                 }
73             });
74     
75             JLabel txtAttenzione = new JLabel("Warning!");
76             txtAttenzione.setIcon(new ImageIcon(getClass().getResource(
77                     Home.URL_IMAGES + "warning16.png")));
78             txtAttenzione.setBounds(10, 85, 100, 30);
79             JLabel txtTesto = new JLabel(
80                     "<html> <ul> <li> image for the banner can not exceed <b> <font color=\"red\">"
81                             + bannerSize.width
82                             + "X"
83                             + bannerSize.height
84                             + "</font></b> pixels."
85                             + "<li> Images of magnitude larger will be resized.</ul></html>");
86             txtTesto.setBounds(0, 90, 420, 80);
87             add(txtTesto, null);
88             add(txtAttenzione, null);
89             add(anteprima, null);
90             add(btnCarica, null);
91         }

111         /**
112          * This method opens the dialog for selecting a file from disk.
113          *
114          * @return
115          * <ul>
116          * <li> Files - the selected file.
117          * <li> Null - if you have not selected any files.
118          * </ul>
119          */
120         private File apriFileDialog() {
121             JFileChooser apriFile = new JFileChooser();
122             apriFile.setDialogTitle("Select a new image");
123             apriFile.setAcceptAllFileFilterUsed(false);
124             apriFile.setMultiSelectionEnabled(false);
125     
126             // File Filter for the window to open the file.
127             apriFile.setFileFilter(new FileFilter() {
128                 public boolean accept(File arg0) {
129                     if (arg0.isDirectory())
130                         return true;
131                     String nome = arg0.getName().toLowerCase();
132                     if (nome.endsWith("jpg") || nome.endsWith("gif")
133                             || nome.endsWith("png"))
134                         return true;
135                     else
136                         return false;
137                 }
138     
139                 public String getDescription() {
140                     return "Images (*.PNG, *.GIF, *.JPG)";
141                 }
142             });
143             int returnVal = apriFile.showOpenDialog(this);
144             if (returnVal == JFileChooser.APPROVE_OPTION) {
145                 return apriFile.getSelectedFile();
146             } else
147                 return null;
148         }
