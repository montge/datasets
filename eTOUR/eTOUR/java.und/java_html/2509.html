172          * Implements the method to obtain statistics about a cultural past
173          * Through Id
174          *
175          * @see unisa.gps.etour.control.GestioneBeniCulturali.IGestioneBeniCulturaliComune#ottieniStatisticheBeneCulturale(int)
176          */
177         public ArrayList<Integer> ottieniStatisticheBeneCulturale(int pBeneCulturaleID) throws RemoteException
178         {
179             /*
180              * This method returns an ArrayList containing 5 elements (0 .. 4).
181              * For each index more 'one is the number of equivalent value your feedback
182              * Index number more 'one.
183              * Even in this case the method is not 'particularly attractive but it does its dirty work
184              * Fine.
185              */
186     
187             if (!ControlloBeniCulturali.controllaIdBeneCulturale(pBeneCulturaleID))
188                 throw new RemoteException(MessaggiErrore.ERRORE_DATI);
189     
190             ArrayList<Integer> listaRisultati = new ArrayList<Integer>(5);
191     
192             // Set all the indices to 0
193             for (int i = 0; i < 5; i++)
194                 listaRisultati.add(i, Integer.valueOf(0));
195     
196             // Calculate the date for the last thirty days
197             Date ultimiTrentaGiorni = new Date(new Date().getTime() - CostantiGlobali.TRENTA_GIORNI);
198     
199             try
200             {
201                 // Get all visits of a certain cultural
202                 // Loop looking for the requests / feedback obtained within the last thirty days
203                 for (BeanVisitaBC b: dbvisita.ottieniListaVisitaBC(pBeneCulturaleID))
204                     if (b.getDataVisita().after(ultimiTrentaGiorni))
205                         listaRisultati.set(b.getVoto() - 1, Integer.valueOf(listaRisultati.get(b.getVoto() - 1).intValue() + 1));
206             }
207             catch (SQLException e)
208             {
209                 throw new RemoteException(MessaggiErrore.ERRORE_DBMS);
210             }
211             catch (Exception e)
212             {
213                 throw new RemoteException(MessaggiErrore.ERRORE_SCONOSCIUTO);
214             }
215     
216             return listaRisultati;
217         }

226         {
227             if (!ControlloBeniCulturali.controllaIdBeneCulturale(pBeneCulturaleID) ||
228                     !ControlloVisiteBeniCulturali.controllaDatiVisitaBeneCulturale(pBeanVisitaBC))
229                 throw new RemoteException(MessaggiErrore.ERRORE_DATI);
230     
231             /*
232              * Please check that the vote has not changed.
233              * If the vote is changed to an exception is raised
234              */
235             boolean votoOk = true;
236     
237             try
238             {
239                 votoOk = dbvisita.ottieniVisitaBC(pBeneCulturaleID, pBeanVisitaBC.getIdTurista()).getVoto() == pBeanVisitaBC.getVoto();
240             }
241             catch (SQLException e)
242             {
243                 throw new RemoteException(MessaggiErrore.ERRORE_DBMS);
244             }
245             catch (Exception e)
246             {
247                 throw new RemoteException(MessaggiErrore.ERRORE_SCONOSCIUTO);
248             }
249     
250             /*
251              * If the vote is not changed we proceed to send the message to the method of
252              * Change the layer's database.
253              */
254             if (votoOk)
255                 try
256                 {
257                     return (dbvisita.modificaVisitaBC(pBeanVisitaBC));
258                 }
259                 catch (SQLException e)
260                 {
261                     throw new RemoteException(MessaggiErrore.ERRORE_DBMS);
262                 }
263                 catch (Exception e)
264                 {
265                     throw new RemoteException(MessaggiErrore.ERRORE_SCONOSCIUTO);
266                 }
