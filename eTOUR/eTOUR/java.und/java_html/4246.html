737         /**
738          * This method opens a tab for the bean tourists supplied input.
739          *
740          * @param pTurista BeanTurista - the bean tourists
741          * @param pModifica boolean <ul> <li> true - if you change the data.
742          * <li> * False - if you receive the card. </ul>
743          * @Return void
744          */
745         private void apriScheda(BeanTurista pTurista, boolean pModifica) {
746             for (int i = 0; i < figli.size(); i++) {
747                 SchedaTurista corrente = figli.get(i);
748                 if (pTurista.getId() == corrente.getId()) {
749                     desktopManager.activateFrame(corrente);
750                     return;
751                 }
752             }
753             SchedaTurista nuova = new SchedaTurista(this, pTurista, pModifica);
754             jDesktopPane.add(nuova, Integer.MAX_VALUE);
755             nuova.setVisible(true);
756             figli.add(nuova);
757         }
