102          *
103          * @param a first value
104          * @param b the second value
105          * @return returns the value calculated using a function T-Norm

107         private static float tNorm(float a, float b) {
108             return ((a * b) / ((2 - ((a + b) - (a * b)))));
109             // Return Math.min (a, b);

111             // Return Math.max (0, a + b-1);
