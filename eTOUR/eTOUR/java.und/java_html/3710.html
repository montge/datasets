10     /**
11      * <b> Punto3DRenderer </b>
12      * <p> This class creates a custom renderer for

14      *
15      * @see javax.swing.table.TableCellRenderer
16      * @see unisa.gps.etour.util.Punto3D
17      * @version 1.0
18      * @author Mario Gallo
19      */
20     public class Punto3DRenderer implements TableCellRenderer {
21         /**
22          * Method that returns the custom component for the
23          * Display of the data contained in the cell of a table.
24          *
25          * @param pTable JTable - the table.
26          * @param pValue Object - the data.
27          * @param pSelected boolean --
28          * <ul>
29          * <li> <i> True </i> if the cell is selected.
30          * <li> <i> False </i> otherwise.
31          * </ul>
32          * @param pHasFocus boolean --
33          * <ul>
34          * <li> <i> True </i> if the cell has the focus.
35          * <li> <i> False </i> otherwise.
36          * </ul>
37          * @param pRow - the line number.
38          * @param pColumn - the column number.
39          * @return Component - the component that customizes render the cell.
40          * @throws IllegalArgumentException - if the value of the cell can not
41          * Be rendered by this renderer.
42          */
43         public Component getTableCellRendererComponent(JTable pTable,
44                 Object pValue, boolean pSelected, boolean pHasFocus, int pRow,
45                 int pColumn) {
46             if (!(pValue instanceof Punto3D)) {
47                 throw new IllegalArgumentException("Value Cella unexpected.");
48             }
49             Punto3D aPoint = (Punto3D) pValue;
50             String point = aPoint.getLatitudine() + ";"
51                     + aPoint.getLongitudine() + ";" + aPoint.getAltitudine();
52             JLabel aLabel = new JLabel(point, SwingConstants.CENTER);
53             if (pSelected) {
54                 aLabel.setForeground(pTable.getSelectionForeground());
55                 aLabel.setBackground(pTable.getSelectionBackground());
56             }
57             return aLabel;
58         }
