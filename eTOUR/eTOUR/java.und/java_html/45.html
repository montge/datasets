275         /**
276          * This method creates the tree starting from the information contained in the two
277          * ArrayList of bean.
278          *
279          * @param ArrayList PPR <BeanPuntoDiRistoro> - the array of places to eat.
280          * @param pBanner ArrayList <BeanBanner> - the array of banners associated.
281          */
282         private void createTree() {
283             // Create the root
284             DefaultMutableTreeNode root = new DefaultMutableTreeNode("eTour");
285     
286             // For each refreshment dell'arraylist calls the method get
287             // Banner.
288             try {
289                 ArrayList<BeanPuntoDiRistoro> pPR = 
290                     gestionePuntiRistoro.ottieniPuntiDiRistoro();
291                 for (int i = 0; i < pPR.size(); i++) {
292                     BeanPuntoDiRistoro corrente = pPR.get(i);
293                     int id = corrente.getId();
294                     PRNode puntoDiRistoro = new PRNode(corrente.getNome(), id);
295                     HashMap<BeanBanner, ImageIcon> banner = gestioneBanner.ottieniBannersDaID(id);
296                     Iterator<BeanBanner> iteraBanner = banner.keySet().iterator();
297                     while (iteraBanner.hasNext()) {
298                         BeanBanner bannercorrente = iteraBanner.next();
299                         BannerNode nuovoBanner = new BannerNode(banner
300                                 .get(bannercorrente), bannercorrente.getId());
301                         puntoDiRistoro.add(nuovoBanner);
302                     }
303     
304                     root.add(puntoDiRistoro);
305                 }
306             } catch (RemoteException ex) {
307                 JLabel error = new JLabel(
308                         "<html><h2>Unable to communicate with the server eTour.</h2>"
309                         + "<h3><u>The list of banners was not loaded.</u></h3>"
310                         + "<p>Please try again later.</p>"
311                         + "<p>If the error persists, please contact technical support.</p>"
312                         + "<p>We apologize for the inconvenience.</html>");
313                 ImageIcon err = new ImageIcon(getClass().getResource(
314                         Home.URL_IMAGES + "error48.png"));
315                 JOptionPane.showInternalMessageDialog(this, error,
316                         "Error!", JOptionPane.ERROR_MESSAGE, err);
317             } finally {
318                 treeBanner.setModel(new DefaultTreeModel(root));
319             }
320         }

515         /**
516          * This method opens the dialog box for entering a new
517          * Banner or editing a banner selected.
518          *
519          * @param pSelectedNode DefaultMutableTreeNode - the selected node.
520          */
521         private void OpenDialog(DefaultMutableTreeNode pSelectedNode) {
522             // This class intercepts mouse events and then
523             // Makes the frame below blocked.
524             class ModalAdapter extends InternalFrameAdapter {
525                 Component glass;
526     
527                 public ModalAdapter(Component pGlassComponent) {
528                     this.glass = pGlassComponent;
529     
530                     MouseInputAdapter adapter = new MouseInputAdapter() {
531                     };
532                     pGlassComponent.addMouseListener(adapter);
533                     pGlassComponent.addMouseMotionListener(adapter);
534                 }
535     
536                 public void internalFrameClosed(InternalFrameEvent s) {
537                     glass.setVisible(false);
538                 }
539             }
540     
541             // Construction of the dialog
542             JOptionPane optionPane = new JOptionPane();
543             final JInternalFrame modal = optionPane.createInternalFrame(
544                     JDesktopPane, "");
545             final JPanel glass = new JPanel();
546             final BannerDialog dialog = new BannerDialog();
547             optionPane.setMessage(dialog);
548             optionPane.setMessageType(JOptionPane.INFORMATION_MESSAGE);
549             JButton[] options = new JButton[2];
550             options[0] = new JButton();
551             options[1] = new JButton("Cancel");
552             options[0].setIcon(new ImageIcon(getClass().getResource(
553                     Home.URL_IMAGES + "Salva16.png")));
554             options[1].setIcon(new ImageIcon(getClass().getResource(
555                     Home.URL_IMAGES + "Annulla16.png")));
556             optionPane.setOptions(options);
557             options[1].addActionListener(new ActionListener() {
558                 public void actionPerformed(ActionEvent arg0) {
559                     modal.setVisible(false);
560                     glass.setVisible(false);
561                 }
562             });
563     
564             glass.setOpaque(false);
565             modal.addInternalFrameListener(new ModalAdapter(glass));
566             glass.add(modal);
567             setGlassPane(glass);
568             modal.setLocation(this.getWidth() / 2, this.getHeight() / 2);
569             glass.setVisible(true);
570             modal.setVisible(true);
571             if (pSelectedNode instanceof BannerNode) // Replace
572             {
573                 final BannerNode banner = (BannerNode) pSelectedNode;
574                 options[0].setText("Replace");
575                 options[0].addActionListener(new ActionListener() {
576                     public void actionPerformed(ActionEvent arg0) {
577                         DefaultTreeModel model = (DefaultTreeModel) treeBanner
578                                 .getModel();
579                         banner.setBanner(dialog.getSelectedBanner());
580                         model.nodeChanged(banner);
581                         glass.setVisible(false);
582                         modal.setVisible(false);
583                     }
584                 });
585     
586                 optionPane.setIcon(new ImageIcon(getClass().getResource(
587                         Home.URL_IMAGES + "SostituisciBanner48.png")));
588                 modal.setTitle("Replace the banner to the point of relief"
589                         + ((PRNode) pSelectedNode.getParent()).getUserObject()
590                                 .toString());
591             } else if (pSelectedNode instanceof PRNode) // Inserting
592             {
593                 final PRNode pr = (PRNode) pSelectedNode;
594                 options[0].setText("Save");
595                 options[0].addActionListener(new ActionListener() {
596                     public void actionPerformed(ActionEvent arg0) {
597                         DefaultTreeModel model = (DefaultTreeModel) treeBanner
598                                 .getModel();
599                         BannerNode nuovo = new BannerNode(dialog
600                                 .getSelectedBanner(), pr.getID());
601                         model.insertNodeInto(nuovo, pr, 0);
602                         glass.setVisible(false);
603                         modal.setVisible(false);
604                     }
605                 });
606                 optionPane.setIcon(new ImageIcon(getClass().getResource(
607                         Home.URL_IMAGES + "nuovoBanner48.png")));
608                 modal.setTitle("Enter banner to the point of relief"
609                         + pSelectedNode.getUserObject().toString());
