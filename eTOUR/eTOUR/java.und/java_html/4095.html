35         public int getScrollableUnitIncrement(Rectangle visibleRect,
36                 int orientation, int direction) {
37     
38             int posCorrente = 0;
39             if (orientation == SwingConstants.HORIZONTAL) {
40                 posCorrente = visibleRect.x;
41             } else {
42                 posCorrente = visibleRect.y;
43             }
44     
45             if (direction < 0) {
46                 int newPos = posCorrente - (posCorrente / maxUnitIncrement)
47                         * maxUnitIncrement;
48                 return (newPos == 0) ? maxUnitIncrement : newPos;
49             } else {
50                 return ((posCorrente / maxUnitIncrement) + 1) * maxUnitIncrement
51                         - posCorrente;
52             }
53         }

55         public int getScrollableBlockIncrement(Rectangle visibleRect,
56                 int orientation, int direction) {
57             if (orientation == SwingConstants.HORIZONTAL) {

59             } else {
60                 return visibleRect.height - maxUnitIncrement;
61             }
