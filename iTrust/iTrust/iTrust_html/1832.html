28     	/**
29     	 * Checks to see if a given date is within a range of months <strong>INCLUSIVELY</strong>, agnostic of
30     	 * the year. <br />
31     	 * <br />
32     	 * 
33     	 * The range "wraps" so that if the first month is after the second month, then the definition of "is in
34     	 * month range" is:<br />
35     	 * the date falls outside of secondMonth, firstMonth, but including secondMonth and firstMonth.
36     	 * 
37     	 * Modular arithmetic is used to adjust month values into the valid range.
38     	 * 
39     	 * @param date
40     	 * @param firstMonth
41     	 * @param secondMonth
42     	 * @return
43     	 */
44     	public static boolean isInMonthRange(java.util.Date date, int firstMonth, int secondMonth) {
45     		Calendar cal = new GregorianCalendar();
46     		cal.setTime(date);
47     		firstMonth %= 12;
48     		secondMonth %= 12;
49     		if (secondMonth >= firstMonth) {
50     			return ((cal.get(Calendar.MONTH) >= firstMonth) && (cal.get(Calendar.MONTH) <= secondMonth));
51     		}
52     		return ((cal.get(Calendar.MONTH) >= firstMonth) || (cal.get(Calendar.MONTH) <= secondMonth));
53     	}

60     	 * @param secondMonth
61     	 * @return
62     	 */
63     	public static boolean currentlyInMonthRange(int firstMonth, int secondMonth) {

108     	 * <!--Pass in INSTANTIATED sql date objects and they will be set to the specified range, ie, FROM
109     	 * <current year> - yearsAgo1/monthValue1/01 TO <current year> - yearsAgo2/monthValue2/<last day of
110     	 * month2>--> Pass in INSTANTIATED sql date objects and they will be set to the specified range, ie, FROM
111     	 * &lt;current year&gt; - yearsAgo1/monthValue1/01 TO &lt;current year&gt; -
112     	 * yearsAgo2/monthValue2/&lt;last day of month2&gt;
113     	 * 
114     	 * @param month1
115     	 *            First sql.Date object to be set
116     	 * @param monthValue1
117     	 * @param yearsAgo1
118     	 * @param month2
119     	 *            Second sql.Date object to be set
120     	 * @param monthValue2
121     	 * @param yearsAgo2
122     	 */
123     	public static void setSQLMonthRange(java.sql.Date month1, int monthValue1, int yearsAgo1,
124     			java.sql.Date month2, int monthValue2, int yearsAgo2) {
125     		GregorianCalendar cal = new GregorianCalendar();
126     		cal.add(Calendar.YEAR, -yearsAgo1);
127     		cal.set(Calendar.MONTH, monthValue1);
128     		cal.set(Calendar.DAY_OF_MONTH, 1);
129     		month1.setTime(cal.getTimeInMillis());
130     		cal.add(Calendar.YEAR, yearsAgo1);
131     		cal.add(Calendar.YEAR, -yearsAgo2);
132     		cal.set(Calendar.MONTH, monthValue2);
133     		cal.set(Calendar.DAY_OF_MONTH, cal.getActualMaximum(Calendar.DAY_OF_MONTH));
134     		month2.setTime(cal.getTimeInMillis());
135     	}
