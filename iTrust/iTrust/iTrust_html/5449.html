51     	 * Check appointments appearing on the calendar for conflicts 
52     	 * with other appointments on the calendar. 
53     	 * 
54     	 * The array from this method is used to determine what appointments
55     	 * will appear in bold on the calendar.
56     	 * 
57     	 * @return An array of items that are in conflict with other items.
58     	 * @throws SQLException
59     	 */
60     	public boolean[] getConflicts() throws SQLException {
61     		boolean conflicts[] = new boolean[send.size()];
62     		for(int i=0; i<send.size(); i++) {
63     			ApptBean ab = send.get(i);
64     			long t = ab.getDate().getTime();
65     			long m = types.getDurationByType(ab.getApptType()) * 60L * 1000L;
66     			Timestamp time = new Timestamp(t+m);
67     			for(int j=i+1; j<send.size(); j++) {
68     				if(send.get(j).getDate().before(time)) {
69     					conflicts[i] = true;
70     					conflicts[j] = true;
71     				}
72     			}
73     		}
74     		return conflicts;
75     	}

87     		List<ApptBean> appts = a_action.getMyAppointments();
88     		Hashtable<Integer, ArrayList<ApptBean>> atable = new Hashtable<Integer, ArrayList<ApptBean>>();
89     		Calendar a = Calendar.getInstance();
90     		for(ApptBean b : appts) {
91     			a.setTimeInMillis(b.getDate().getTime());
92     			if(a.get(Calendar.MONTH) == thisMonth && a.get(Calendar.YEAR) == thisYear) {
93     				if(!atable.containsKey(a.get(Calendar.DAY_OF_MONTH)))
94     					atable.put(a.get(Calendar.DAY_OF_MONTH), new ArrayList<ApptBean>());
95     				ArrayList<ApptBean> l = atable.get(a.get(Calendar.DAY_OF_MONTH));
96     				l.add(b);
97     				send.add(b);
98     				atable.put(a.get(Calendar.DAY_OF_MONTH), l);
99     			}
100     		}
101     		return atable;
102     	}
