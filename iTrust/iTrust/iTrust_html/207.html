144     	 * @return the combined list
145     	 */
146     	private List<HospitalBean> combineLists(List<HospitalBean> hcpsHospitals, List<HospitalBean> allHospitals) {
147     		for (HospitalBean hos : allHospitals) {
148     			if (!hcpsHospitals.contains(hos))

150     		}
151     		return hcpsHospitals;

185     	public String hasInteraction(String drug, String pid, String startdate, String enddate) throws iTrustException{
186     		String response = "";
187     		try {
188     
189     			SimpleDateFormat original = new SimpleDateFormat("MM/dd/yyyy");
190     			SimpleDateFormat needed = new SimpleDateFormat("yyyy/MM/dd");
191     			
192     			Date start = original.parse(startdate);
193     			startdate = needed.format(start);
194     			Date end = original.parse(enddate);
195     			enddate = needed.format(end);
196     			
197     			
198     			startdate = startdate.replaceAll("/", "-");
199     			enddate = enddate.replaceAll("/", "-");
200     			List<PrescriptionReportBean> prBeanList = ovDAO.getPrescriptionReportsByDate(Long.parseLong(pid), startdate, enddate);
201     			List<DrugInteractionBean> dBeanList = drugDAO.getInteractions(drug);
202     			for (PrescriptionReportBean prBean : prBeanList){
203     				String presDrug = prBean.getPrescription().getMedication().getNDCode();
204     				for (DrugInteractionBean dBean : dBeanList){
205     					String intDrug1 = dBean.getFirstDrug();
206     					String intDrug2 = dBean.getSecondDrug();
207     					
208     					if (presDrug.equals(intDrug1) && drug.equals(intDrug2)){
209     						response += "Currently Prescribed: " + prBean.getPrescription().getMedication().getDescription() + 
210     						". Start Date: " + prBean.getPrescription().getStartDateStr() 
211     						+ ", End Date: " + prBean.getPrescription().getEndDateStr() + ". ";
212     						response += "Interactions: " + prBean.getPrescription().getMedication().getDescription() + " - " + ndcDAO.getNDCode((dBean.getSecondDrug())).getDescription() + ". ";
213     						response += "Description: " + dBean.getDescription() + "  ";
214     					} else if (presDrug.equals(intDrug2) && drug.equals(intDrug1)){
215     						response += "Currently Prescribed: " + prBean.getPrescription().getMedication().getDescription() + ". Start Date: " + prBean.getPrescription().getStartDateStr() 
216     						+ ", End Date: " + prBean.getPrescription().getEndDateStr() + ". ";
217     						response += "Interactions: " + prBean.getPrescription().getMedication().getDescription() + " - " + ndcDAO.getNDCode(dBean.getFirstDrug()).getDescription() + ". ";
218     						response += "Description: " + dBean.getDescription() + "  ";
219     					}
220     				}
221     			}
222     
223     		} catch (DBException e){
224     			e.printStackTrace();
225     			throw new iTrustException(e.getMessage());
226     		} catch (ParseException e){
227     			e.printStackTrace();
228     			throw new iTrustException(e.getMessage());
229     		}
230     		return response;
231     	}

234     		String response = "";
235     		try {
236     			List<AllergyBean> allergyList = allergyDAO.getAllergies(Long.parseLong(patMID));
237     			if (allergyList.isEmpty()) return response;
238     			MedicationBean medBean = ndcDAO.getNDCode(ndcode);
239     			if(medBean == null) return response;
240     			String newDrug = medBean.getDescription();
241     			for (AllergyBean allergyBean : allergyList){
242     				String currentAllerDrug = allergyBean.getDescription();
243     				//Allergy: Aspirin. First Found: 12/20/2008. 
244     				if (newDrug.equals(currentAllerDrug)){
245     					response += "Allergy: " + newDrug + ". First Found: " + allergyBean.getFirstFoundStr();
246     				}
247     			}
248     		} catch (DBException e){
249     			e.printStackTrace();
250     			throw new iTrustException(e.getMessage());
251     		}
252     		return response;
253     	}

289     			FormValidationException {
290     		if (form.getAddMedID() != null && !"".equals(form.getAddMedID())) {
291     			new EditOfficeVisitValidator(true).validate(form);
292     			PrescriptionBean pres = new PrescriptionBean();
293     			pres.setDosage(Integer.valueOf(form.getDosage()));

295     			pres.setStartDateStr(form.getStartDate());
296     			pres.setInstructions(form.getInstructions());
297     			MedicationBean med = new MedicationBean();
298     			med.setNDCode(form.getAddMedID());
299     			pres.setMedication(med);
300     			pres.setVisitID(ovID);
301     			transDAO.logTransaction(TransactionType.ADD_PRESCRIPTION, loggedInMID, pid, "EditOffceVisit - Add prescription - " + pres.getMedication().getNDCode());
302     			ovDAO.addPrescription(pres);
303     		}

327     	 * Adds a diagnosis or a procedure to an office visit
328     	 * 
329     	 * @param action the type of action to add
330     	 * @param code the CPT code of the action
331     	 * @param visitID the office visit to add the action to
332     	 * @param dateOfDeath the date of death, if needed
333     	 * @return true if the operation completed; false if the code was null
334     	 * @throws DBException
335     	 * @throws iTrustException
336     	 */
337     	
338     	private boolean checkAddSubAction(OVSubAction action, String code, long visitID, String dateOfDeath)
339     			throws DBException, iTrustException {
340     		if (code == null || "".equals(code)) {
341     			return false;
342     		} else {
343     			switch (action) {
344     			case ADD_DIAGNOSIS:
345     				ovDAO.addDiagnosisToOfficeVisit(Double.valueOf(code), visitID);
346     				break;
347     			case ADD_IMMUNIZATION:
348     			case ADD_PROCEDURE:
349     				ovDAO.addProcedureToOfficeVisit(code, visitID, "");
350     				break;
351     			case ADD_LAB_PROCEDURE:
352     				ovDAO.addLabProcedureToOfficeVisit(code, visitID, pid);
353     				break;
354     			default:
355     				return false;
356     			}
357     			return true;
358     		}
359     	}

372     	 * Removes an action from an office visit
373     	 * 
374     	 * @param action type of action to remove
375     	 * @param input id for the action to remove
376     	 * @return true if the operation completed; false if the code was null
377     	 * @throws DBException
378     	 */
379     
380     	private boolean checkRemoveSubAction(OVSubAction action, String input) throws DBException {
381     		if (input == null || "".equals(input))
382     			return false;
383     		long removeID;
384     		try {
385     			removeID = Long.valueOf(input);
386     		} catch (NumberFormatException e) {
387     			e.printStackTrace();
388     			return false;
389     		}
390     		switch (action) {
391     		case REMOVE_DIAGNOSIS:
392     			ovDAO.removeDiagnosisFromOfficeVisit(removeID);
393     			break;
394     		case REMOVE_IMMUNIZATION:
395     		case REMOVE_PROCEDURE:
396     			ovDAO.removeProcedureFromOfficeVisit(removeID);
397     			break;
398     		case REMOVE_MEDICATION:
399     			ovDAO.removePrescription(removeID);
400     			break;
401     		case REMOVE_LAB_PROCEDURE:
402     			ovDAO.removeLabProcedureFromOfficeVisit(removeID);
403     			break;
404     		}
405     		return true;
406     	}
