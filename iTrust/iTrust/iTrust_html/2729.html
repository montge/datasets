3     import java.io.IOException;

7     /**
8      * This exception is used specifically for when an action involves the correct
9      * and proper submission of a web form.  Form Validation is handled by a series of
10      * other classes, but when form validation is incorrect or incomplete, this exception
11      * is thrown. 
12      */
13     public class FormValidationException extends Exception {
14     	private static final long serialVersionUID = 1L;
15     	private ErrorList errorList;
16     
17     	/**
18     	 * Constructor with error messages passed as a list of parameters to the method.
19     	 * @param errorMessages The list of error messages to be returned in the special form validation box.
20     	 */
21     	public FormValidationException(String... errorMessages) {
22     		errorList = new ErrorList();
23     		for (String msg : errorMessages) {
24     			errorList.addIfNotNull(msg);
25     		}
26     	}
27     
28     	/**
29     	 * Constructor with error messages as a special ErrorList data type.
30     	 * @param errorList An ErrorList object which contains the list of error messages.
31     	 */
32     	public FormValidationException(ErrorList errorList) {
33     		this.errorList = errorList;
34     	}
35     
36     	/**
37     	 * Returns the error list as a java.util.List of Strings.
38     	 * @return The error list
39     	 */
40     	public List<String> getErrorList() {
41     		return errorList.getMessageList();
42     	}
43     
44     	/**
45     	 * The error message will be displayed at the top of the iTrust page as in other iTrust Exceptions.
46     	 */
47     	@Override
48     	public String getMessage() {
49     		return "This form has not been validated correctly. The following field are not properly filled in: "
50     				+ errorList.toString();
51     	}
52     
53     	/**
54     	 * The special formatting for error messages is then kept in one place.
55     	 * @param out The output writer (in this case a JSPWriter) where the formatted list will go.
56     	 * @throws IOException If the writer is incorrect.
57     	 */
58     	public void printHTML(JspWriter out) throws IOException {
59     		out.print("<h2>Information not valid</h2><div class=\"errorList\">");
60     		for (String errorMessage : errorList) {
61     			out.print(errorMessage + "<br />");
62     		}
63     		out.print("</div>");
64     	}
65     }
