66     	/**
67     	 * Returns a list of all the HCPs who have a particular diagnosis
68     	 * 
69     	 * @param icdcode the diagnosis of interest
70     	 * @return the list of HCPs
71     	 * @throws DBException
72     	 */
73     	public List<HCPDiagnosisBean> getHCPByDiagnosis(String icdcode) throws DBException {
74     		
75     		int medMatch = 0;
76     		HashMap<Long, HCPDiagnosisBean> hcpHash = new HashMap<Long, HCPDiagnosisBean>();
77     		HashMap<Long, Long> patientHash = new HashMap<Long, Long>();
78     	
79     		transactionDAO.logTransaction(TransactionType.FIND_HCPS_WITH_EXP, loggedInMID);
80     		
81     		HCPDiagnosisBean diag = null;
82     		List<OfficeVisitBean> beans = officeVisitDAO.getAllOfficeVisitsForDiagnosis(icdcode);
83     		
84     		for (OfficeVisitBean bean: beans) {
85     			
86     			// check for HCP-Patient locality based on first 3 digits in ZIP
87     			if (!patientDAO.getPatient(loggedInMID).getZip1().substring(0, 2).
88     					equals(personnelDAO.getPersonnel(bean.getHcpID()).getZip1().substring(0, 2)))
89     				continue;
90     			
91     			// Check to see if we already have a bean for the HCP associated with this visit
92     			if (hcpHash.containsKey(bean.getHcpID())) {
93     				diag = (HCPDiagnosisBean)hcpHash.get(bean.getHcpID());
94     				
95     				for (PrescriptionBean p: bean.getPrescriptions()) {
96     					List<MedicationBean> mlist = diag.getMedList();
97     					
98     					for (MedicationBean b: mlist) {
99     						if (p.getMedication().getDescription().equals(b.getDescription()))
100     							medMatch++;
101     					}
102     					if (medMatch == 0) {
103     						mlist.add(p.getMedication());
104     						diag.setMedList(mlist);
105     					}
106     					else {
107     						medMatch = 0;
108     					}
109     				}
110     				
111     				// Get Lab Procedures
112     				List<LabProcedureBean> labprocs = diag.getLabList(); 
113     				List<LabProcedureBean> lpbeans = labprocDAO.getAllLabProceduresForDocOV(bean.getVisitID());
114     				for (LabProcedureBean p: lpbeans) {
115     					labprocs.add(p);
116     				}
117     				diag.setLabList(labprocs);
118     				
119     				
120     				if (surveyDAO.isSurveyCompleted(bean.getVisitID())) {
121     					SurveyBean survey = surveyDAO.getSurveyData(bean.getVisitID());
122     					diag.setVisitSat(survey.getVisitSatisfaction());
123     					diag.setTreatmentSat(survey.getTreatmentSatisfaction());
124     				}
125     				
126     				// Check if this patient has been seen multiple times for this diagnosis
127     				if (!patientHash.containsKey(bean.getPatientID())) {
128     					patientHash.put(bean.getPatientID(), bean.getHcpID());
129     					diag.incNumPatients();
130     				}
131     			}
132     			else {
133     				diag = new HCPDiagnosisBean();
134     				List<MedicationBean> mlist = new ArrayList<MedicationBean>();
135     				diag.setHCP(bean.getHcpID());
136     				try {
137     					diag.setHCPName(personnelDAO.getName(bean.getHcpID()));
138     				} catch (iTrustException e) {
139     					diag.setHCPName("null");
140     				}
141     				diag.incNumPatients();
142     				for (PrescriptionBean p: bean.getPrescriptions()) {
143     					mlist.add(p.getMedication());
144     				}
145     				diag.setMedList(mlist);
146     				diag.setLabList(labprocDAO.getAllLabProceduresForDocOV(bean.getVisitID()));
147     				
148     				if (surveyDAO.isSurveyCompleted(bean.getVisitID())) {
149     					SurveyBean survey = surveyDAO.getSurveyData(bean.getVisitID());
150     					diag.setVisitSat(survey.getVisitSatisfaction());
151     					diag.setTreatmentSat(survey.getTreatmentSatisfaction());
152     				}
153     				
154     				patientHash.put(bean.getPatientID(), bean.getHcpID());
155     				hcpHash.put(bean.getHcpID(), diag);
156     			}
157     		}
158     		List<HCPDiagnosisBean> list = new ArrayList<HCPDiagnosisBean>(hcpHash.values());
159     		Collections.sort(list, new HCPDiagnosisBeanComparator() );
160     		return list;
161     	}

163     	/**
164     	 * Looks up all the prescriptions given by a certain HCP with the same ICD code.
165     	 * @param hcpid The MID of the HCP
166     	 * @param icdcode The ICD code of the prescription we are looking up.
167     	 * @return A java.util.List of PrescriptionBeans made by this HCP of this ICD code.
168     	 * @throws DBException
169     	 */
170     	public List<PrescriptionBean> getPrescriptionsByHCPAndICD(long hcpid, String icdcode) throws DBException {
171     		List<PrescriptionBean> list = new ArrayList<PrescriptionBean>();
172     		
173     		List<OfficeVisitBean> ovs = officeVisitDAO.getAllOfficeVisitsForDiagnosis(icdcode);
174     		for (int i = 0; i < ovs.size(); i++) {
175     			if (ovs.get(i).getHcpID() == hcpid) {
176     				list.addAll(ovs.get(i).getPrescriptions());
177     			}
178     		}
179     		
180     		return list;
181

187     	 * Checks to see what HCP has had the most experience with a diagnosis
188     	 *
189     	 */
190     	static class HCPDiagnosisBeanComparator implements Comparator<HCPDiagnosisBean>, Serializable {
191     		
192     		private static final long serialVersionUID = -6328390386684022934L;
193     
194     		/**
195     		 * Compares one HCP with another
196     		 * 
197     		 * @param a the first HCP
198     		 * @param b the second HCP
199     		 * @return -1 if a has had more patients, 1 if b has had more patients; otherwise 0
200     		 */
201     		public int compare(HCPDiagnosisBean a, HCPDiagnosisBean b) {
202     			int ret = 0;
203     			
204     			if (a.getNumPatients() > b.getNumPatients())
205     				ret = -1;
206     			else if (a.getNumPatients() < b.getNumPatients())
207     				ret = 1;
208     			
209     			return ret;
210     		}
211     	}
