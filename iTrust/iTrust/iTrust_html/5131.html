151     		Connection conn = null;

153     		try {
154     			conn = factory.getConnection();
155     			ps = conn.prepareStatement("INSERT INTO HCPAssignedHos (HCPID, HosID) VALUES (?,?)");
156     			ps.setLong(1, hcpID);
157     			ps.setString(2, hospitalID);
158     			return (1 == ps.executeUpdate());
159     		} catch (SQLException e) {
160     			if (1062 == e.getErrorCode())
