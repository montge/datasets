43     	 * Checks to see if a user exists with the given mid

52     				return 0;

62     	 * Checks to see if the number of reset password attempts has been exceeded for the given ipAddress

73     	 * Checks if the given mid matches the given role
74     	 * 
75     	 * @param mid
76     	 *            the mid to be checked

79     	 * @return true if the mid and role match
80     	 * @throws iTrustException
81     	 */
82     	public String checkRole(long mid, String role) throws iTrustException {
83     		try {
84     			if (("patient".equals(role) && patientDAO.getRole(mid, role).equals("patient"))
85     					|| ("hcp".equals(role) && patientDAO.getRole(mid, role).equals("hcp"))
86     					|| ("uap".equals(role) && patientDAO.getRole(mid, role).equals("uap"))
87     					|| ("pha".equals(role) && patientDAO.getRole(mid, role).equals("pha"))
88     					|| ("er".equals(role) && patientDAO.getRole(mid, role).equals("er")))
89     				return role;
90     			else
91     				return null;
92     		} catch (DBException e) {
93     			e.printStackTrace();
94     		} catch (iTrustException e) {
95     			e.printStackTrace();

133     	/**
134     	 * Resets the password for the given mid
135     	 * 
136     	 * @param mid of the user to have their password reset
137     	 * @param role what role the user has in iTrust
138     	 * @param answer answers to their security question
139     	 * @param password their password
140     	 * @param confirmPassword their password again
141     	 * @param ipAddr the ip address the request is coming from
142     	 * @return status message
143     	 * @throws FormValidationException
144     	 * @throws DBException
145     	 * @throws iTrustException
146     	 */
147     	public String resetPassword(long mid, String role, String answer, String password,
148     			String confirmPassword, String ipAddr) throws FormValidationException, DBException,
149     			iTrustException {
150     
151     		Role r = authDAO.getUserRole(mid);
152     		try {
153     			Role.parse(role);
154     		} catch (IllegalArgumentException e) {
155     			return "Invalid role";
156     		}
157     
158     		if (r.equals(Role.ADMIN))
159     			return "This role cannot be changed here";
160     		if (!r.equals(Role.parse(role)))
161     			return "Role mismatch";
162     
163     		if (authDAO.getResetPasswordFailures(ipAddr) >= MAX_RESET_ATTEMPTS) {
164     			return "Too many retries";
165     		}
166     
167     		try {
168     			validatePassword(password, confirmPassword);
169     
170     			if (answer.equals(authDAO.getSecurityAnswer(mid))) {
171     				authDAO.resetPassword(mid, password);
172     				new EmailUtil(factory).sendEmail(makeEmailApp(mid, role));
173     				return "Password changed";
174     				
175     			} else {
176     				authDAO.recordResetPasswordFailure(ipAddr);
177     				return "Answer did not match";
178     			}
179     
180     		} catch (DBException e) {
181     			return "Error in validation of security answer";
182     		}
183     	}

186     	 * Creates and sends an e-mail about the change
187     	 *

202     
203     			return email;
204     		}
205     		else{ //UAP or HCP - admin taken out in "resetPassword"
206     			PersonnelBean p = new PersonnelDAO(factory).getPersonnel(mid);
207     			Email email = new Email();
208     			email.setFrom("no-reply@itrust.com");
209     			email.setToList(Arrays.asList(p.getEmail()));
210     			email.setSubject("Your password has been changed in iTrust");
211     			email.setBody(String.format("Dear %s, \n You have chosen to change your iTrust password for user %s", p.getFullName(), mid));
212     
213     			return email;
214     		}
215     	}

217     	/**
218     	 * Checks to make sure the password is correctly entered twice.
219     	 * 
220     	 * @param password the password
221     	 * @param confirmPassword the password again for confirmation
222     	 * @throws FormValidationException
223     	 */
224     
225     	private void validatePassword(String password, String confirmPassword) throws FormValidationException {
226     		ErrorList errorList = new ErrorList();
227     		if (password == null || "".equals(password)) {
228     			errorList.addIfNotNull("Password cannot be empty");
229     		} else {
230     			if (!password.equals(confirmPassword))
231     				errorList.addIfNotNull("Passwords don't match");
232     			if (!ValidationFormat.PASSWORD.getRegex().matcher(password).matches()) {
233     				errorList.addIfNotNull("Password must be in the following format: "
234     						+ ValidationFormat.PASSWORD.getDescription());
235     			}
236     		}
237     		if (errorList.hasErrors())
238     			throw new FormValidationException(errorList);
239     	}
