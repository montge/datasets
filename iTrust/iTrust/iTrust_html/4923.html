26      * added. DAOs can assume that all data has been validated and is correct.

55     	 * @throws iTrustException
56     	 * @throws DBException
57     	 */
58     	public String getName(long mid) throws iTrustException, DBException {
59     		Connection conn = null;
60     		PreparedStatement pstmt = null;
61     		try {
62     			conn = factory.getConnection();
63     			pstmt = conn.prepareStatement("SELECT firstName, lastName FROM Personnel WHERE MID=?");
64     			pstmt.setLong(1, mid);
65     			ResultSet rs;
66     
67     			rs = pstmt.executeQuery();
68     			if (rs.next()) {
69     				return rs.getString("firstName") + " " + rs.getString("lastName");
70     			} else {
71     				throw new iTrustException("User does not exist");
72     			}
73     		} catch (SQLException e) {
74     			e.printStackTrace();
75     			throw new DBException(e);
76     		} finally {
77     			DBUtil.closeConnection(conn, pstmt);
78     		}
