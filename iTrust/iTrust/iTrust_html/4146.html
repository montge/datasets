168     	 * Set a given HCP as declared
169     	 * 
170     	 * @param name HCP to declare
171     	 * @return An empty string.
172     	 * @throws iTrustException
173     	 */
174     	public String declareHCP(String name) throws iTrustException {
175     		boolean match = false;
176     		for (HCPVisitBean visit: visits) {
177     			if (0 == visit.getHCPName().toLowerCase().compareTo(name.toLowerCase())) {
178     				match = true;
179     				Long mid = Long.valueOf(visit.getHCPMID());
180     				if (!patientDAO.checkDeclaredHCP(patientMID, visit.getHCPMID())) {
181     					declareAction.declareHCP(mid.toString());
182     				}
183     				visit.setDesignated(true);
184     			}
185     		}
186     		
187     		if (!match) {
188     				List<PersonnelBean> doclist = docDAO.getAllPersonnel();
189     				for (PersonnelBean ele: doclist) {
190     					if (0 == name.compareTo(ele.getFullName())) {
191     						HCPVisitBean visitBean;
192     						visitBean = new HCPVisitBean();
193     						visitBean.setHCPMID(ele.getMID());
194     						visitBean.setHCPName(ele.getFullName());
195     						visitBean.setOVDate("");
196     						visitBean.setHCPSpecialty(ele.getSpecialty());
197     						visitBean.setHCPAddr(ele.getStreetAddress1() +" "+ ele.getStreetAddress2() +" "+ ele.getCity() +", "+ ele.getState() +" "+ ele.getZip());
198     						
199     						visitBean.setDesignated(true);
200     				
201     						Long mid = Long.valueOf(ele.getMID());
202     						if (!patientDAO.checkDeclaredHCP(patientMID, mid)) {
203     							declareAction.declareHCP(mid.toString());
204     							visits.add(visitBean);
205     						}
206     					}
207     				}
208     			
209     		}
210     		return "";
211     	}

243     							if (null != zip && !zip.equals("")) {

251     						}

254     						if (ele.getZip().contains(zip.substring(0, 2))) {
