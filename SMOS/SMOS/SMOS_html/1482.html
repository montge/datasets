83     	 * tabella.
84     	 * 
85     	 * @param pField
86     	 *            Il campo di cui vogliamo recuperare il valore massimo.
87     	 * @param pTable
88     	 *            La tabella in cui ricercare le informazioni richieste.

90     	 *         parametro, presente nella tabella pTable, passata come parametro.
91     	 * 
92     	 * @throws EntityNotFoundException
93     	 * @throws ConnectionException
94     	 * @throws SQLException
95     	 */
96     	synchronized static public int getMaxValue(String pField, String pTable)
97     			throws EntityNotFoundException, ConnectionException, SQLException {
98     
99     		int value = 0;
100     		Connection connect = null;
101     		try {
102     			/*
103     			 * Se non sono stati forniti il campo e la tabella restituiamo un
104     			 * codice di errore
105     			 */
106     			if (pField.equals(""))
107     				throw new EntityNotFoundException();
108     			if (pTable.equals(""))
109     				throw new EntityNotFoundException();
110     
111     			/*
112     			 * Prepariamo la stringa SQL per recuperare le informazioni
113     			 * richieste
114     			 */
115     			String tSql = "SELECT max(" + pField + ") as new_field FROM "
116     					+ pTable;
117     
118     			// Otteniamo una Connessione al DataBase
119     			connect = DBConnection.getConnection();
120     			if (connect == null)
121     				throw new ConnectionException();
122     
123     			// Inviamo la Query al DataBase
124     			ResultSet tRs = Utility.queryOperation(connect, tSql);
125     
126     			if (tRs.next())
127     				value = tRs.getInt("new_field");
128     			else
129     				throw new EntityNotFoundException();
130     
131     			return value;
132     		} finally {
133     			DBConnection.releaseConnection(connect);
134     		}
135     	}

463     	 * @param sourceStr
464     	 * @param searchFor
465     	 * @param replaceWith
466     	 * @return la stringa corrispondente a searchBuffer
467     	 */
468     	public static String ReplaceAll(String sourceStr, String searchFor,
469     			String replaceWith) {
470     		StringBuffer searchBuffer = new StringBuffer(sourceStr);
471     		int hits = 0;
472     
473     		while (searchBuffer.toString().toUpperCase().indexOf(
474     				searchFor.toUpperCase(), hits) >= 0) {
475     			int newIndex = searchBuffer.toString().toUpperCase().indexOf(
476     					searchFor.toUpperCase(), hits);
477     			searchBuffer.replace(newIndex, newIndex + searchFor.length(),
478     					replaceWith);
479     			hits++;
480     		}
481     
482     		return searchBuffer.toString();
483     	}

488     	/**
489     	 * @param sourceStr
490     	 * @param searchFor
491     	 * @param replaceWith
492     	 * @return newStringBuffer
493     	 */
494     	public static String ReplaceAllStrings(String sourceStr, String searchFor,
495     			String replaceWith) {
496     		StringBuffer searchBuffer = new StringBuffer(sourceStr);
497     		StringBuffer newStringBuffer = new StringBuffer("");
498     
499     		while (searchBuffer.toString().toUpperCase().indexOf(
500     				searchFor.toUpperCase()) >= 0) {
501     			int newIndex = searchBuffer.toString().toUpperCase().indexOf(
502     					searchFor.toUpperCase());
503     			newStringBuffer.append(searchBuffer.substring(0, newIndex));
504     			newStringBuffer.append(replaceWith);
505     			searchBuffer = new StringBuffer(searchBuffer.substring(newIndex
506     					+ searchFor.length(), searchBuffer.length()));
507     		}
508     
509     		newStringBuffer.append(searchBuffer);
510     		return newStringBuffer.toString();
511     	}

791     	public static String getActualDate() {
792     		GregorianCalendar gc = new GregorianCalendar();
793     
794     		String date="";
795     		int year=gc.get(GregorianCalendar.YEAR);
796     		
797     		int month=gc.get(GregorianCalendar.MONTH)+1;
798     		String months="";
799     		if(month<10){
800     			months="0"+month;
801     		}else{
802     			months= months+month;
803     		}
804     		
805     		int day=gc.get(GregorianCalendar.DAY_OF_MONTH);
806     		String days="";
807     		if(day<10){
808     			days="0"+day;
809     		}else{
810     			days= days+day;
811     		}
812     		
813     		date= date + days+"/"+months+"/"+year;
814     	
815     		return date;
816     		
817     	}
