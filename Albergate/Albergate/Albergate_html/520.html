234     	boolean dataInizioIsNotBeforeToday()
235     	{
236     		Date inizio_pren = DateUtils.convertDate(data_inizio.getText());
237     		Date today = new Date();
238     		
239     		if (inizio_pren.getYear() < today.getYear())
240     			return false;
241     		else
242     			if (inizio_pren.getYear() > today.getYear())
243     				return true;
244     			else // l'anno e' il medesimo
245     				if (inizio_pren.getMonth() < today.getMonth())
246     					return false;
247     				else
248     					if (inizio_pren.getMonth() > today.getMonth())
249     						return true;
250     					else // l'anno e il mese sono i medesimi
251     						if (inizio_pren.getDate() < today.getDate())
252     							return false;
253     						else
254     							return true;
255     	}

261     	void ctrlAvvio()
262     	{
263     		Frame msg;
264     
265     		if (Utils.dateEPeriodoCorretti(this, data_inizio.getText(), data_fine.getText()))
266     		{
267     			if (con_ag.getState() && nome_ag.getText().equals(""))
268     			{
269     				msg = new MessageDialog(this, " Manca il nome dell'agenzia! ");
270     				return;
271     			}
272     			
273     			ListaDisponibilita L_disp = (Principale.db).elencoDisponibilita();
274     			if (L_disp == null)
275     			{
276     				msg = new MessageDialog(this, " Problemi con il database nella lettura delle disponibilita'! ");
277     				return;
278     			}
279     			if (caller == 3)
280     			{	
281     				if (Utils.data1MaggioreData2(DateUtils.convertDate(data_fine.getText()), save_date) )
282     				{	
283     					msg = new AvvisoDialog(this, "ATTENZIONE: la data non puo' essere posticipata");
284     					data_fine.setText(DateUtils.parseDate(DateUtils.giveStringOfDate(save_date)));
285     				}
286     				else
287     					if (Utils.data1MinoreData2(DateUtils.convertDate(data_fine.getText()), DateUtils.convertDate(data_inizio.getText()) ) )
288     					{
289     						msg = new AvvisoDialog(this, "ATTENZIONE: la data non puo' essere precedente alla data di inizio");
290     						data_fine.setText(DateUtils.parseDate(DateUtils.giveStringOfDate(save_date)));			
291     					}
292     					else
293     					{
294     						stanze_disp = new ListaStanze();
295     						startAvvio(L_disp, 0);
296     					}
297     			}
298     			else
299     			{
300     				if (dataInizioIsNotBeforeToday())
301     				{
302     					stanze_disp = new ListaStanze();
303     					startAvvio(L_disp, 0);
304     				}
305     				else
306     					msg = new MessageDialog(this, " La data di inizio deve essere maggiore o uguale di quella odierna! ");
307     			}
308     		}
309     	}

327     	void startAvvio(ListaDisponibilita L_disp, int k)
328     	{
329     		if (con_ag.getState())
330     			startSearchComm(L_disp, k);
331     		
332     		// le date sono corrette se si arriva qui in quanto controllate in ctrlAvvio()
333     		Frame msg;
334     		String str = new String("");
335     		ListaCommissionamenti L_comm = new ListaCommissionamenti();
336     		ListaStanze elenco_stanze_disp = new ListaStanze();
337     
338     		// variabili necessarie per passaggio dei parametri della Inserisci
339     		data1 = DateUtils.parseDate(data_inizio.getText());
340     		data2 = DateUtils.parseDate(data_fine.getText());
341     
342     		// scansione delle disponibilita
343     		Stanza stanza = null;
344     		Disponibilita disp_attuale = null;
345     		for (int i = 1; i <= L_disp.length(); i++)
346     		{
347     			disp_attuale = L_disp.getDisponibilita(i);
348     			if ( disp_attuale.isDisponibile( DateUtils.convertDate(data_inizio.getText()), DateUtils.convertDate(data_fine.getText()), Flag.DISPONIBILE) )
349     			{
350     				stanza = (Principale.db).readStanza(disp_attuale.getNumStanza());
351     				if (stanza != null)
352     				{
353     					elenco_stanze_disp.addStanza(stanza);
354     					stanze_disp.addStanza(stanza);
355     				}
356     				else
357     					msg = new MessageDialog(this, " Problemi con il database! ");
358     			}
359     		}
360     		
361     		if (!stanze_disp.isEmpty())
362     		{
363     			Avvio.setEnabled(false);
364     			Annulla.setEnabled(false);
365     			data_inizio.setEditable(false);
366     			data_fine.setEditable(false);
367     			Annulla1.setEnabled(true);
368     			Conferma.setEnabled(true);
369     			con_ag.setEnabled(false);
370     			nome_ag.setEditable(false);
371     			for (int i = 1; i <= elenco_stanze_disp.length(); i++)
372     				lista.addItem(str+(elenco_stanze_disp.getStanza(i)).toString());
373     		}
374     		else
375     			if (k == 0)
376     				if (!con_ag.getState())
377     					msg = new AvvisoDialog(this, " Nessuna stanza disponibile in tale periodo! ");
378     				else
379     					msg = new AvvisoDialog(this, " Nessuna stanza disponibile o commisionata in tale periodo! ");	
380     	}

382     	void startSearchComm(ListaDisponibilita L_disp, int k)
383     	{
384     		// le date sono corrette se si arriva qui in quanto controllate in ctrlAvvio()
385     		Frame msg;
386     		
387     		Commissionamento c;
388     		Stanza stanza = null;
389     		Disponibilita disp_loc = null;
390     		ListaCommissionamenti L_comm = new ListaCommissionamenti();;
391     		ListaStanze elenco_stanze_disp = new ListaStanze();
392     		Date d1 = DateUtils.convertDate(data_inizio.getText());
393      		Date d2 = DateUtils.convertDate(data_fine.getText());
394     		
395     		data1 = DateUtils.parseDate(data_inizio.getText());
396     		data2 = DateUtils.parseDate(data_fine.getText());
397     		L_comm = (Principale.db).foundCommissionamentiSenzaLike(nome_ag.getText());
398     		if (L_comm == null)
399     		{
400     			msg = new MessageDialog(this, " Problemi con il database nella lettura delle commissioni! ");
401     			return;
402     		}
403     		if ( !L_comm.isEmpty() )
404     		{
405     			nomi_agenzie = new String[L_comm.length()];
406     			int indice_nomi = 0;
407     			for (int i = 1; i <= L_comm.length(); i++)
408     			{
409     				c = L_comm.getCommissionamento(i);
410     				// Controllo che le date inserite appartengano al periodo commissionato
411     				if( (!Utils.data1MaggioreData2(c.getInizioComm(), d1)) && (!(c.getFineComm()).before(d2))) 
412     				{
413     					disp_loc = Utils.getDispOfRoom(L_disp, c.getNumStanza(), false);
414     					if (disp_loc.isDisponibile(d1, d2, Flag.COMMISSIONATA))
415     					{
416     						stanza = (Principale.db).readStanza(disp_loc.getNumStanza());
417     						if (stanza != null)
418     						{	
419     							elenco_stanze_disp.addStanza(stanza);
420     							stanze_disp.addStanza(stanza);
421     							nomi_agenzie[indice_nomi] = new String(c.getNomeAgenzia());
422     							indice_nomi++;
423     						}
424     						else
425     						{
426     							msg = new MessageDialog(this, " Problemi con il database! ");
427     							return;
428     						}
429     					}
430     				}
431     			}
432     		}	
433     		if (!stanze_disp.isEmpty())
434     		{
435     			Avvio.setEnabled(false);
436     			Annulla.setEnabled(false);
437     			data_inizio.setEditable(false);
438     			data_fine.setEditable(false);
439     			Annulla1.setEnabled(true);
440     			Conferma.setEnabled(true);
441     			for (int i = 1; i <= elenco_stanze_disp.length(); i++)
442     				lista.addItem("COMMISSIONATA A "+nomi_agenzie[i-1]+" "+(elenco_stanze_disp.getStanza(i)).toString());
443     		}
444     	} // startSearchComm
