107     	protected Stanza readDatiStanza()
108     	{
109     		Stanza room = new Stanza();
110     		room.setNumStanza(testo1.getText());
111     		room.setPostiLetto(Integer.parseInt(testo2.getText()));
112     		Float f = Float.valueOf(testo3.getText());
113     		room.setPxBase(f.floatValue());
114     		room.setDescrizione(testo4.getText());
115     		if 	(mycheckboxes.getState())
116     			room.setDispLettoAgg(Const.SI);
117     		return room;	 
118     	}
119     
120     	protected void writeDatiStanza(Stanza room)
121     	{
122     		testo1.setText(room.getNumStanza());
123     		testo2.setText(""+room.getPostiLetto());
124     		testo3.setText(""+room.getPxBase());
125     		testo4.setText(room.getDescrizione());    	
126     		if ( room.getDispLettoAgg() == Const.SI)
127     			mycheckboxes.setState(true);
128     		else
129     			mycheckboxes.setState(false);	
130     	}

141     	protected boolean errori()
142     	{
143     		Frame msg;
144     		
145     		if ( !((testo1.getText()).length() <= 4) )
146     		{
147     			msg = new MessageDialog(this, " Il numero della stanza deve essere composto da al piu' 4 caratteri! ");
148     			return true;	
149     		}
150     		if ( ((testo1.getText()).equals("")) )
151     		{
152     			msg = new MessageDialog(this, " Manca il numero della stanza! ");
153     			return true;	
154     		}
155     		if ( ((testo2.getText()).equals("")) )
156     		{
157     			msg = new MessageDialog(this, " Manca il numero di posti letto! ");
158     			return true;	
159     		}
160     		if ( !Utils.isIntPos(testo2.getText()) )
161     		{
162     			msg = new MessageDialog(this, " Il numero di posti deve essere un numero positivo! ");
163     			return true;	
164     		}
165     		if ( ((testo3.getText()).equals("")) )
166     		{
167     			msg = new MessageDialog(this, " Manca il prezzo della stanza! ");
168     			return true;	
169     		}
170     		if ( !Utils.isFloatPos(testo3.getText()) )
171     		{
172     			msg = new MessageDialog(this, " Il prezzo della stanza deve essere un numero positivo! ");
173     			return true;	
174     		}
175     		if ( !((testo4.getText()).length() <= 52) )
176     		{
177     			msg = new MessageDialog(this, " La descrizione deve essere composta da al piu' 52 caratteri! ");
178     			return true;
