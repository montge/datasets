61     					if ( (stanza.getPostiLetto() == num_sogg) && (stanza.getDispLettoAgg() == Const.SI) )

63     							
64     				(Principale.db).changePrenotazione(prenotazione.getIdPrenotazione(), prenotazione.getNumStanza(), 
65     														prenotazione.getNumPers(),prenotazione.getNome(),prenotazione.getCognome(), 
66     														prenotazione.getNumTel(), prenotazione.getInizioSogg(), prenotazione.getFineSogg(), prenotazione.getDataPren(),
67     														prenotazione.getCaparra(), prenotazione.getRichLettoAgg(), prenotazione.getPensionamento(),
68     														prenotazione.getTramiteAgenzia(), prenotazione.getNomeAgenzia(), prenotazione.getRichParticolari());	
69     				msg = new AvvisoLocale(this, " Nuovo soggiornante registrato! ");
70                	}
