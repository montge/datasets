160             void aggiornaAvideo(boolean addebito)
161             {
162             		Frame msg;
163                     int supp[];
164     
165                     BeneServizio extra_scelto = elenco_extra.getBeneServizio( extra.getSelectedIndex() + 1 );
166                     int pos;
167                     if ((pos = posExtraInAddebiti(extra_scelto)) < 0 && addebito)
168                     {
169                             Addebito a = new Addebito("RIST", extra_scelto.getCodExtra(), 0, 0);
170                             elenco_addebiti.addAddebito( a );
171                             elenco_extra_addebitati.addBeneServizio( extra_scelto );
172     
173                             // allungo l'array delle quantita' modificate
174                             supp = new int[elenco_addebiti.length()];
175                             for (int i = 0; i<quantita_mod.length; i++)
176                                     supp[i] = quantita_mod[i];
177                             supp[supp.length - 1] = 1;
178                             quantita_mod = supp;
179                             pos = quantita_mod.length;
180                             extra_add.setText(extra.getSelectedItem());
181                             continuaAggiornamento();
182                     }
183                     else
184                     {
185                             if (pos < 0)
186                             {
187                                     // sto cercando di togliere da un addebito inesistente
188                                     msg = new AvvisoDialog(this, "Stai cercando di togliere da un addebito ormai inesistente!");
189                             }
190                             else
191                             {
192                                     // aggiorno tale array per contenere l'addebito appena effettuato
193                                     if (addebito)
194                                     {
195                                             quantita_mod[pos - 1]++;
196                                             extra_add.setText(extra.getSelectedItem());
197                                             continuaAggiornamento();
198                                     }
199                                     else
200                                             if ( quantita_mod[pos - 1] > 0)
201                                             {
202                                                     quantita_mod[pos - 1]--;
203                                                     extra_add.setText(extra.getSelectedItem());
204                                                     continuaAggiornamento();
205                                             }
206                                             else
207     			                                msg = new AvvisoDialog(this, "Stai cercando di togliere da un addebito ormai inesistente!");
208                             }
209                     }
210             }

218                     
219                     if (addebiti.getItemCount() > 0)

222                     {

226     	                        conto_di_questo_extra = quantita_mod[i-1]*e.getPxUnitario();
227         	                    if ( Integer.parseInt(e.getCodExtra()) >= 40000000 && Integer.parseInt(e.getCodExtra()) < 50000000)
228             	                        addebiti.addItem("S "+Utils.completaStringa(e.getDescrizione(), 32)+" "+(Principale.config).getValuta()+"."+(elenco_addebiti.getAddebito(i)).getTotAddebito());
229                 	            else if ( Integer.parseInt(e.getCodExtra()) >= 50000000)
230                     	                addebiti.addItem("R "+Utils.completaStringa(e.getDescrizione(), 32)+" "+(Principale.config).getValuta()+"."+(elenco_addebiti.getAddebito(i)).getTotAddebito());
231                         	    else
232                             	        addebiti.addItem( Utils.completaStringa(e.getDescrizione(),10)+Utils.completaStringa(" x "+quantita_mod[i-1], 5)+(Principale.config).getValuta()+"."+conto_di_questo_extra );
233     	                        conto = conto + conto_di_questo_extra;
234         	                }
235        	                    i++;
236                     }
237                     totale.setText((Principale.config).getValuta()+"."+conto);
238             }

250     		private void writeFile()
251     		{
252     			Frame msg;
253     			float costo_coperti;
254     		
255     			DataOutputStream outStream;
256     			try
257     			{	
258     				outStream = new DataOutputStream(new FileOutputStream("conto.abg"));
259     			}
260     			catch (IOException ex)
261     			{
262     				msg = new MessageDialog(this,"Errore nell'apertura del file del conto");
263     				return;
264     			}
265     			try
266     			{
267     				outStream.writeBytes( (Principale.config).getNomeAlbergo()+"\n" );
268     				outStream.writeBytes( "\n" );
269     				outStream.writeBytes( (Principale.config).getRagSoc()+"\n" );
270     				outStream.writeBytes( (Principale.config).getIndirizzoAlb()+"\n" );
271     				outStream.writeBytes( (Principale.config).getComuneAlb()+"\n" );
272     				outStream.writeBytes( (Principale.config).getCittaAlb()+"\n" );
273     				outStream.writeBytes( "\n" );
274     
275     				BeneServizio extra;
276     				Addebito a;				
277     				int num;
278     				String supporto;
279     				float conto_totale;
280     				
281     				for(int i=1; i<=elenco_addebiti.length(); i++)
282     				{
283     					if (quantita_mod[i-1] != 0)
284     					{
285     						a = elenco_addebiti.getAddebito(i);
286     						extra = elenco_extra_addebitati.getBeneServizio(i);
287     						num = quantita_mod[i-1];
288     						if ( (extra.getCodExtra()).compareTo("40000000") < 0 )
289     							supporto = new String(extra.getDescrizione()+" x "+num+" ");
290     						else if ( (extra.getCodExtra()).compareTo("50000000") < 0 )
291     							supporto = new String("Supplemento per "+extra.getDescrizione()+" ");
292     						else
293     							supporto = new String("Riduzione per "+extra.getDescrizione()+" ");
294     						while (supporto.length() < 52)
295     							supporto = new String(supporto+" ");
296     						outStream.writeBytes(supporto+(Principale.config).getValuta()+"."+
297     												num*extra.getPxUnitario()+"\n" );
298     					}
299     				}
300     				supporto = new String("Coperto x "+num_coperti.getText());
301     				while (supporto.length() < 52)
302     					supporto = new String(supporto+" ");
303     				costo_coperti = Integer.parseInt(num_coperti.getText())*(Principale.config).getCoperto();
304     				outStream.writeBytes(supporto+(Principale.config).getValuta()+"."+costo_coperti+"\n");
305     				outStream.writeBytes("\n");
306     				supporto = new String("Totale");
307     				while (supporto.length() < 52)
308     					supporto = new String(supporto+" ");
309     				conto_totale = (Float.valueOf( new String((totale.getText()).substring(2)) )).floatValue()+costo_coperti;
310     				outStream.writeBytes( supporto+(Principale.config).getValuta()+"."+conto_totale );
311     				outStream.close();
312     			}	
313     			catch (IOException ex)
314     			{
315     				msg = new MessageDialog(this,"Errore nella scrittura del file del conto");	
316     				return;
317     			}			
318     		}
